74  {

81  void append(double val) noexcept
82  {
83  ++size_;
84  double delta{val - mean_};
85  mean_ += delta / size_;
86  sum2_ += delta * (val - mean_);
87  min_ = std::min(min_, val);
88  max_ = std::max(max_, val);
89  }

92  {
93  append(first);

95  append(args...);
