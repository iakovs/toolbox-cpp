85template <typename FnT>
86std::size_t parse_frame(ConstBuffer buf, FnT fn)
87{
88 std::size_t consumed{0};
89 for (;;) {
90 const auto* data = buffer_cast<const char*>(buf);
91 const std::size_t size = buffer_size(buf);
92 if (size < sizeof(std::uint32_t)) {
93 break;
94 }
95 const auto total = sizeof(std::uint32_t) + get_length(data);
96 if (size < total) {
97 break;
98 }
99 fn(ConstBuffer{data + sizeof(std::uint32_t), total - sizeof(std::uint32_t)});
100 buf = advance(buf, total);
101 consumed += total;
102 }
103 return consumed;
104}
