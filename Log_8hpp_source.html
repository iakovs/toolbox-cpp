35 class Log {

37  friend Log& operator<<(Log& log, ValueT&& val)
38  {
39  log.os_ << std::forward<ValueT>(val);
40  return log;
41  }

43  public:
44  explicit Log(WallTime ts, LogLevel level) noexcept
45  : ts_{ts}

50  }
