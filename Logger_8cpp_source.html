56 class StdLogger final : public Logger {
57  void do_write_log(WallTime ts, LogLevel level, LogMsgPtr&& msg, size_t size) noexcept override
58  {
59  const auto t{WallClock::to_time_t(ts)};
60  tm tm;
61  localtime_r(&t, &tm);
62 
63  // The following format has an upper-bound of 49 characters:
64  // "%Y/%m/%d %H:%M:%S.%06d %-7s [%d]: "
65  //
66  // Example:
67  // 2022/03/14 00:00:00.000000 WARNING [0123456789]: msg...
68  // <---------------------------------------->
69  char head[49 + 1];
70  size_t hlen{strftime(head, sizeof(head), "%Y/%m/%d %H:%M:%S", &tm)};
71  const auto us{static_cast<int>(us_since_epoch(ts) % 1000000)};
72  hlen += sprintf(head + hlen, ".%06d %-7s [%d]: ", us, log_label(level),
73  static_cast<int>(gettid()));
74  char tail{'\n'};
75  iovec iov[] = {
76  {head, hlen}, //
77  {msg.get(), size}, //
78  {&tail, 1} //
79  };
80 
81  int fd{level > LogLevel::Error ? STDOUT_FILENO : STDERR_FILENO};
82  // The following lock was required to avoid interleaving.
83  lock_guard<mutex> lock{mutex_};
84  // Best effort given that this is the logger.
85  ignore = writev(fd, iov, sizeof(iov) / sizeof(iov[0]));
86  }
87  mutex mutex_;
88 } std_logger_;

90 class SysLogger final : public Logger {
91  void do_write_log(WallTime ts, LogLevel level, LogMsgPtr&& msg, size_t size) noexcept override
92  {
93  int prio;
94  switch (level) {
95  case LogLevel::None:
96  return;
97  case LogLevel::Crit:
98  prio = LOG_CRIT;
99  break;
100  case LogLevel::Error:
101  prio = LOG_ERR;
102  break;
103  case LogLevel::Warning:
104  prio = LOG_WARNING;
105  break;
106  case LogLevel::Notice:
107  prio = LOG_NOTICE;
108  break;
109  case LogLevel::Info:
110  prio = LOG_INFO;
111  break;
112  default:
113  prio = LOG_DEBUG;
114  }
115  syslog(prio, "%.*s", static_cast<int>(size), static_cast<const char*>(msg.get()));
116  }
