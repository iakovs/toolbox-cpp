118  void put_integer(char c)
119  {
120  if (sign_ == 0) {
121  if (c == '+') {
122  sign_ = 1;
123  return;
124  }
125  if (c == '-') {
126  sign_ = -1;
127  return;
128  }
129  // Imply positive sign.
130  sign_ = 1;
131  }
132  if (c == '\r') {
133  // Ignore.
134  return;
135  }
136  if (c != '\n') {
137  if (c < '0' || c > '9') {
138  // Fatal protocol exception.
139  throw Exception{"invalid integer"};
140  }
141  num_ = num_ * 10 + (c - '0');
142  return;
143  }
144  flush([this]() { static_cast<DerivedT*>(this)->on_resp_integer(sign_ * num_); });
145  }
146  void put_bulk_string(char c)
147  {
148  // TODO: add support for null bulk strings.
149  if (sign_ == 0) {
150  switch (c) {
151  case '\r':
152  // Ignore.
153  break;
154  case '\n':
155  sign_ = 1;
156  break;
157  default:
158  if (c < '0' || c > '9') {
159  // Fatal protocol exception.
160  throw Exception{"invalid length"};
161  }
162  num_ = num_ * 10 + (c - '0');
163  break;
164  }
165  return;
166  }
167  if (num_ > 0) {
168  tok_ += c;
169  --num_;
170  return;
171  }
172  // End of line.
173  if (c == '\r') {
174  // Ignore.
175  return;
176  }
177  if (c != '\n') {
178  // Fatal protocol exception.
179  throw Exception{"invalid bulk string"};
180  }
181  flush([this]() { static_cast<DerivedT*>(this)->on_resp_string(tok_); });
182  }
183  void put_array(char c)
184  {
185  if (c == '\r') {
186  // Ignore.
187  return;
188  }
189  if (c != '\n') {
190  if (c < '0' || c > '9') {
191  // Fatal protocol exception.
192  throw Exception{"invalid length"};
193  }
194  num_ = num_ * 10 + (c - '0');
195  return;
196  }
197  bool ok{false};
198  int popped{0};
199  if (num_ > 0) {
200  stack_.push(num_);
201  } else {
202  popped = 1 + pop_if_end();
203  }
204  const auto finally = make_finally([&]() noexcept {
205  if (!ok) {
206  if (!is_top_level()) {
207  bad_ = true;

212  }
213  }
214  clear_tok();
215  });
216  // NOLINTNEXTLINE(readability-braces-around-statements)
217  if (bad_) [[unlikely]] {
218  if (is_top_level() && popped > 0) {

220  static_cast<DerivedT*>(this)->on_resp_reset(); // noexcept
221  }
222  } else {
223  static_cast<DerivedT*>(this)->on_resp_array_begin(num_);
